#include <bits/stdc++.h> 
void bubbleSort(vector<int>& arr, int n)
{   
    // Write your code here.
    for( int i = 0 ; i<arr.size()-1 ; i++ ){
        for( int j = 0 ; j<arr.size()-i-1 ; j++ ){
            if(arr[j]>arr[j+1]){
                swap(arr[j],arr[j+1]);
            }
        }
    }
}
